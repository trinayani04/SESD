  <script>
    // small client-side registration persistence (local-only)
    function handleRegister(e){
      e.preventDefault();
      const name=document.getElementById('name').value.trim();
      const email=document.getElementById('email').value.trim();
      const cat=document.getElementById('category').value;
      const interest=document.getElementById('interest').value;
      if(!name||!email){document.getElementById('regMsg').textContent='Please fill name & email';return}
      // basic local-store
      const regs=JSON.parse(localStorage.getItem('aurora_regs')||'[]');
      regs.push({name,email,cat,interest,ts:new Date().toISOString()});
      localStorage.setItem('aurora_regs',JSON.stringify(regs));
      document.getElementById('regMsg').textContent='Registration saved locally — check your dashboard for details.';
      document.getElementById('regForm').reset();
    }

    // progressive enhancement: show number of registrants if any
    (function(){
      const count=JSON.parse(localStorage.getItem('aurora_regs')||'[]').length;
      if(count>0){
        const p=document.createElement('p');p.className='muted';p.style.fontSize='13px';p.style.marginTop='10px';p.textContent=`${count} registrations so far (local)`;
        document.querySelector('.hero-card').appendChild(p);
      }
    })();
  </script>
